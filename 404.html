    static func Click-here(&)
    {
        this::before()->color("blue")
        this::active()->color("green")
        if(Console.input.MouseClick(0))
        {
                print("connecting...")
                scan(user.input(href) , x)
                try
                {
                if(Console.getResponse(x) as HttpWebResponse)
                {
                    print("connection successful, redirecting...")
                    Console.openURL(x)
                }
                else
                {
                    throw "ERROR, COULDN'T ACCESS WEBSITE!"
                    return
                }
                }
        }
    }
